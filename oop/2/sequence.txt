@startuml
actor User
actor Admin
boundary SocialNetwork

User -> SocialNetwork : Sign Up
SocialNetwork -> User : Confirmation

User -> SocialNetwork : Log In
SocialNetwork -> User : JWT Token

User -> SocialNetwork : Create Post
SocialNetwork -> User : Post Confirmation

User -> SocialNetwork : View Posts
SocialNetwork -> User : Display Posts

User -> SocialNetwork : Report Post
SocialNetwork -> Admin : Notify Admin of Report

Admin -> SocialNetwork : Moderate Report
SocialNetwork -> Admin : Post Moderated (Delete/Edit)

@enduml
